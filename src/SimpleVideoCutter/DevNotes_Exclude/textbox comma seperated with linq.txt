 private void ConvertButton_Click(object sender, EventArgs e)
        {
            // Get the input text from the TextBox
            string inputText = textBox1.Text;

            // Split the input text by commas and remove empty entries using LINQ
            List<string> resultList = inputText
                .Split(',')
                .Select(entry => entry.Trim())  // Trim to remove leading/trailing spaces
                .Where(entry => !string.IsNullOrWhiteSpace(entry))
                .ToList();

            // Display the result as a comma-separated string
            string resultText = string.Join(", ", resultList);

            // Show the result in a MessageBox
            MessageBox.Show("Result: " + resultText);
        }
